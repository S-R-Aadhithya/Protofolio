 54def ingest_linkedin():
 55    """
 56    Scrapes a provided LinkedIn profile URL to extract the headline and adds it to the user's semantic memory.
 57
 58    ### How to Use
 59    ```javascript
 60    fetch('/api/ingest/linkedin', { method: 'POST', body: JSON.stringify({linkedin_url: "https://linkedin.com/in/..."}) })
 61    ```
 62
 63    ### Why this function was used
 64    Gathers professional headline and presence context to inform the target job role during deliberation.
 65
 66    ### How to change in the future
 67    Since basic scraping is highly brittle against LinkedIn, swap the `requests`/`bs4` block for an official API or a specialized proxy service like Proxycurl.
 68
 69    ### Detailed Line-by-Line Execution
 70    - Line 1: `url = (request.json or {}).get('linkedin_url', '').strip()` -> Safely pulls the URL.
 71    - Line 2: `if 'linkedin.com/in/' not in url: return jsonify({"error": "Invalid URL"}), 400` -> Fast validation against incorrect formats.
 72    - Line 3: `u = User.query.filter_by(email=get_jwt_identity()).first(); u.linkedin_handle = url.rstrip('/').split('/')[-1]; db.session.commit()` -> Extracts handle, updates the database, and commits immediately.
 73    - Line 4: `try: text = BeautifulSoup(requests.get(url, headers={"User-Agent": "Mozilla/5.0"}, timeout=10).text, 'html.parser').find('h2').get_text(strip=True) if SCRAPE_AVAILABLE else f"Profile: {url}"\nexcept Exception: text = f"Profile: {url}"` -> Soup scraping with graceful fallback if the module is missing or the CSS selector misses.
 74    - Line 5: `MemoryManager().add_fact(user_id=u.email, content=Chairman().process_ingestion(f"Handle: {u.linkedin_handle}. {text}", "linkedin"))` -> Mints the fact into Mem0.
 75    - Line 6: `return jsonify({"message": "Done", "handle": u.linkedin_handle}), 200` -> Resolves response.
 76
 77    Returns: 
 78        flask.Response: JSON response containing handle.
 79    """
 80    url = (request.json or {}).get('linkedin_url', '').strip()
 81    if 'linkedin.com/in/' not in url: return jsonify({"error": "Invalid URL"}), 400
 82    u = User.query.filter_by(email=get_jwt_identity()).first()
 83    u.linkedin_handle = url.rstrip('/').split('/')[-1]
 84    db.session.commit()
 85    try: text = BeautifulSoup(requests.get(url, headers={"User-Agent": "Mozilla/5.0"}, timeout=10).text, 'html.parser').find('h2').get_text(strip=True) if SCRAPE_AVAILABLE else f"Profile: {url}"
 86    except Exception: text = f"Profile: {url}"
 87    MemoryManager().add_fact(user_id=u.email, content=Chairman().process_ingestion(f"Handle: {u.linkedin_handle}. {text}", "linkedin"))
 88    return jsonify({"message": "Done", "handle": u.linkedin_handle}), 200

 90@ingest_bp.route('/github', methods=['POST'])
 91@jwt_required()
 92def fetch_github():
 93    """
 94    Leverages PyGithub to hit the GitHub API, grabbing the user's top 10 starred repositories and passing them into Mem0.
 95
 96    ### How to Use
 97    Assuming the user previously OAuth'd, just send a blank POST to `/api/ingest/github`.
 98
 99    ### Why this function was used
100    To ground the AI generations in real codebase metrics and project names, elevating the portfolio's authenticity.
101
102    ### How to change in the future
103    You can increase the repo depth from 10 to 50, but consider the token context size limits in Mem0.
104
105    ### Detailed Line-by-Line Execution
106    - Line 1: `u = User.query.filter_by(email=get_jwt_identity()).first()` -> Identifies caller.
107    - Line 2: `if not u or not u.github_access_token: return jsonify({"error": "No GitHub token"}), 400` -> Guards against null tokens.
108    - Line 3: `try: gh = Github(u.github_access_token); repos = sorted([{"name": r.name, "description": r.description, "language": r.language, "stars": r.stargazers_count, "url": r.html_url} for r in gh.get_user().get_repos(type="public")], key=lambda x: x['stars'] or 0, reverse=True)[:10]` -> Connects to PyGithub and maps/sorts/slices the user's repos in a single hyper-dense comprehension list.
109    - Line 4: `Summary = Chairman().process_ingestion(f"Top repos: {json.dumps(repos)}", "github")` -> Summarizes JSON blob synthetically.
110    - Line 5: `db.session.add(MemoryChunk(user_id=u.id, content=Summary, metadata_json=json.dumps({"source": "github"}))); db.session.commit()` -> Backs up raw chunk to SQL.
111    - Line 6: `MemoryManager().add_fact(user_id=u.email, content=Summary); return jsonify({"repos": repos}), 200` -> Embeds in vector DB and responds.
112    - Line 7: `except Exception as e: return jsonify({"error": str(e)}), 500` -> Catch-all firewall for API failures.
113
114    Returns: 
115        flask.Response: JSON describing repos parsed.
116    """
117    u = User.query.filter_by(email=get_jwt_identity()).first()
118    if not u or not u.github_access_token: return jsonify({"error": "No GitHub token"}), 400
119    try:
120        gh = Github(u.github_access_token)
121        repos = sorted([{"name": r.name, "description": r.description, "language": r.language, "stars": r.stargazers_count, "url": r.html_url} for r in gh.get_user().get_repos(type="public")], key=lambda x: x['stars'] or 0, reverse=True)[:10]
122        Summary = Chairman().process_ingestion(f"Top repos: {json.dumps(repos)}", "github")
123        db.session.add(MemoryChunk(user_id=u.id, content=Summary, metadata_json=json.dumps({"source": "github"})))
124        db.session.commit()
125        MemoryManager().add_fact(user_id=u.email, content=Summary)
126        return jsonify({"repos": repos}), 200
127    except Exception as e:
128        return jsonify({"error": str(e)}), 500
